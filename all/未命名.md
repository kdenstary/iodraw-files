import matplotlib.pyplot as plt
import networkx as nx
from matplotlib.font_manager import FontProperties

# 设置中文字体
font = FontProperties(fname='C:\Windows\Fonts\simhei.ttf')  # 替换为你的中文字体路径

# 创建空图
G = nx.DiGraph()

# 添加节点和边
G.add_node('关系型数据库授权使用服务')
G.add_nodes_from(['数据存储与检索', '事务处理', '数据一致性与完整性', '数据安全与访问控制', '数据备份与恢复', '高可用性'])

# 各功能的细分服务
services = {
    '数据存储与检索': ['高效存储结构\n(表、索引、视图)', 'SQL查询优化'],
    '事务处理': ['ACID属性', '并发控制'],
    '数据一致性与完整性': ['ACID属性'],
    '数据安全与访问控制': ['权限分配\n与角色管理', '加密存储\n与传输'],
    '数据备份与恢复': ['定期自动备份', '灾难恢复机制'],
    '高可用性': ['负载均衡', '多节点集群']
}

# 添加服务之间的连线
G.add_edges_from([
    ('关系型数据库授权使用服务', '数据存储与检索'),
    ('关系型数据库授权使用服务', '事务处理'),
    ('关系型数据库授权使用服务', '数据一致性与完整性'),
    ('关系型数据库授权使用服务', '数据安全与访问控制'),
    ('关系型数据库授权使用服务', '数据备份与恢复'),
    ('关系型数据库授权使用服务', '高可用性')
])

# 添加各功能的细分服务
for key, values in services.items():
    for value in values:
        G.add_edge(key, value)

# 画图
pos = nx.spring_layout(G)
plt.figure(figsize=(12, 8))
nx.draw(G, pos, with_labels=True, node_size=3000, node_color="skyblue", font_size=10, font_weight="bold", font_color="black", font_family=font.get_name(), arrows=True, arrowstyle="->", arrowsize=15)

# 显示图像
plt.title('关系型数据库授权使用服务示意图', size=15, fontproperties=font)
plt.show()
